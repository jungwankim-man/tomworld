{{- /* Render top-N posts by cumulative PV from data/views.json. Filters to /posts/* paths only, joins each path to its Page so we can render title + permalink. Silently no-ops when data is missing or there are no eligible posts. */ -}} {{- $top := 5 -}} {{- $views := site.Data.views | default dict -}} {{- /* Build [{path, pv}] excluding _meta and non-post paths. */ -}} {{- $items := slice -}} {{- range $path, $pv := $views -}} {{- if and (ne $path "_meta") (hasPrefix $path "/posts/") (ne $path "/posts/") (gt (printf "%v" $pv | int) 0) -}} {{- $items = $items | append (dict "path" $path "pv" (printf "%v" $pv | int)) -}} {{- end -}} {{- end -}} {{- /* Sort desc by pv, take top N, resolve each to a Page. */ -}} {{- $items = sort $items "pv" "desc" -}} {{- $items = first $top $items -}} {{- $resolved := slice -}} {{- range $items -}} {{- $p := site.GetPage .path -}} {{- if $p -}} {{- $resolved = $resolved | append (dict "page" $p "pv" .pv) -}} {{- end -}} {{- end -}} {{- with $resolved }}
