If still empty, render nothing (silent no-op) */ -}} {{- $top := 3 -}} {{- $cur := . -}} {{- $related := slice -}} {{- /* Pool from shared categories. */ -}} {{- range .GetTerms "categories" -}} {{- range .Pages -}} {{- if and (ne . $cur) (eq .Section "posts") -}} {{- $related = $related | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- /* If short, top up from shared tags. */ -}} {{- if lt (len $related) $top -}} {{- range .GetTerms "tags" -}} {{- range .Pages -}} {{- if and (ne . $cur) (eq .Section "posts") (not (in $related .)) -}} {{- $related = $related | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Dedupe by RelPermalink (a post may appear via both cat and tag). */ -}} {{- $seen := slice -}} {{- $dedup := slice -}} {{- range $related -}} {{- if not (in $seen .RelPermalink) -}} {{- $seen = $seen | append .RelPermalink -}} {{- $dedup = $dedup | append . -}} {{- end -}} {{- end -}} {{- /* Sort by date desc, take top N. */ -}} {{- $picks := first $top (sort $dedup "Date" "desc") -}} {{- with $picks }}
